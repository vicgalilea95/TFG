\chapter{Medida y dimensión de Hausdorff}
\section{Preliminares}
\begin{defi}
\rm
Sea X un conjunto arbitrario, $\Sigma$ una colección de subconjuntos de X. Se dice que $\Sigma$ es una $\sigma$-álgebra en X si:
	
	1-	$\emptyset \in \Sigma$
	
	2- $A\in\Sigma\Longrightarrow A^c\in\Sigma$
	
	3- $\{A_i\}_{i\in\mathbb{R}}\subseteq\Sigma\Rightarrow\cup_{i\in\mathbb{N}}\{A_i\}\in\Sigma$

\end{defi}


Al par ($X, \Sigma$) se le llama espacio medible.


\begin{defi}
\rm
Sea X un conjunto arbitrario.$\mathcal{A}_0\subseteq\mathcal{P}(X)$ se le dice semianillo, si cumple.
	
	1-	A, B$\in\mathcal{A}_0$ $ \Rightarrow$ A$\cap$ B $\in X$
	
	2- A, B$\in\mathcal{A}_0$ $ \Rightarrow\exists\{A_i\}_{i=1}^k$ t.q. $ A-B=\cup_{i=1}^kA_i$
	

\end{defi}
\begin{defi}
\rm
 Sea $(X, \Sigma)$ un espacio medible. Una medida positiva sobre un semianillo $\mathcal{A}_0\subseteq\mathcal{P}(X)$ es una función: 
$\mu:\mathcal{A}_0\longmapsto[0,\infty]$ tal qué,

\begin{itemize}
    \item $\mu(\emptyset)=0$
    \item Si $\{A_i\}_{i=1..n}$ son disjuntos dos a dos, y su unión se encuentra en el semianillo, entonces:\[\mu(\bigcup_{i=1}^n\{A_i\})=\sum_{i=1}^n\mu({A_i})\]
\end{itemize}




\end{defi}
Notar que la estructura de semianillo no garantiza que la unión de dos elementos del semianillo, esté en él semianillo.

\begin{defi}
\rm
Sea $(X, \Sigma)$ un espacio medible. Una medida positiva en este espacio es una función: 

$\mu:\Sigma\longmapsto[0,\infty]$ tal qué,

1-$\mu(\emptyset)=0$

2-Si $\{A_i\}_{i\in\mathbb{N}}$ son disjuntos dos a dos, entonces 
$\Rightarrow\mu(\cup_{i\in\mathbb{N}}\{A_i\})=\sum_{i\in\mathbb{N}}\mu({A_i})$


\end{defi}
A la terna $(X,\Sigma, \mu)$ se le llama espacio de medida. Y este es completo sii:
\begin{equation*}
B \in \Sigma : \mu(B)=0 \Rightarrow \forall A \in \Sigma \wedge A \subset B, \mu(A)=0
\end{equation*}

Frecuentemente, hay definida una medida $\mu$ sobre un conjunto pequeño de conjuntos y queremos extenderla a una familia más grande de conjuntos. Esta situación motiva la definición de medida exterior.
\begin{defi}
\rm
Sea X y el conjunto de partes de X, p(X), una medida exterior en X es una función del tipo:
\begin{equation*}
\mu^*:p(X)\longmapsto[0,\infty]
\end{equation*}
tal que:



1-$\mu(\emptyset)=0$

2-Si $A \subseteq B \Rightarrow \mu^*(A) \leqslant\mu^*(B)$

3-Si $\{A_i\}_{i\in\mathbb{N}}$ es una sucesión de subconjuntos, entonces 
$\Rightarrow\mu^*(\cup_{i\in\mathbb{N}}\{A_i\})\leqslant\sum_{i\in\mathbb{N}}\mu^*({A_i})$


\end{defi}
\begin{prop}
Sea $\mathcal{C}\subseteq\mathcal{P}(X), \emptyset\in\mathcal{C}$ y $\rho:\mathcal{C}\longrightarrow [0, \infty]$ tal que $\rho(\emptyset)=0, \forall B \subseteq X$ se define:
\begin{equation*}
\mu^*(B):= inf \lbrace\sum_{n=1}^\infty \rho(A_n) : A_n \in \mathcal{C}, B\subset \bigcup_{n=1}^\infty A_n \rbrace
\end{equation*}
Entonces:

1-$\mu^*$ es una medida exterior tal qué $\mu^*(A) \leq \rho(A), \forall A \in \mathcal{C}$, que la llamaremos medida exterior generada por $\rho$.

2-Si $\mathcal{C}$ es un semianillo y $\rho$ una medida generada por el semianillo, $\rho$ y $\mu^*$ coinciden en $\mathcal{C}$.
\end{prop}

\title{Teorema de extensión de Caratheodory}

Notar que una medida exterior no tiene restricciones en cuanto a los subconjuntos sobre los que está definida, mientras que la medida positiva necesitaba estar definida sobre una $\sigma$-algebra.
Aunque la medida exterior es más débil que la positiva en el sentido de que no es numerablemente aditiva. Ahora el objetivo es encontrar una $\sigma$-algebra, donde las medidas exteriores, si sean numerablemente aditivas.
\begin{defi}
\rm
Sea $\mu^*$ una medida exterior en X. A$\subseteq$X diremos que es $\mu^*$-medible si $\forall B\subseteq X$
\begin{equation}
\mu^{*}(B)=\mu(B\cap A)^*+\mu(B\cap A^c)
\end{equation}
\end{defi}
Denotaremos $M_{\mu^*}$ a la familia de conjuntos $\mu^*$-medibles.
\begin{teo}[De extensión de Caratheodory]
Sea X un conjunto y $\mu^*$ una medida exterior en X. Entonces:

a)$M_{\mu^*}$ es una $\sigma$-algebra.

b)La restricción de $\mu$ a $M_{\mu^*}$ es una medida.

c)$(X, M_{\mu^*}, \mu^*)$ es completo.

d)Si $\mu^*$ es la medida exterior generada por la medida sobre un semianilllo $\rho$ de un semianillo $\mathcal{A}_0 \Rightarrow \mathcal{A}_0 \subset M_{\mu^*}$

\end{teo}

\begin{teo}[Teorema de Hahn]
 Toda medida $\sigma$-finita en un semianillo $\mathcal{A}_0$ se extiende de forma única a cada $\sigma$-álgebra entre $A_0$ y $ M_{\mu^*}$.
\end{teo}
\begin{ej}[Medida de Lebesgue en $\mathbb{R}$]


Sea $\mathcal{C} = \lbrace (a,b] : a \leq b \rbrace$ y $\rho (a,b]=b-a$,

Veamos que $\mathcal{C}$ cumple las propiedades de semianillo:

1-Dados dos intervalos semi-abiertos A, B su intersección o es vacía o es otro intervalo semi-abierto.

2-La resta de intervalos semi-abiertos es otro intervalo semi-abierto

Se puede ver que $\rho$ es una medida sobre el semianillo. Ahora definamos la medida exterior de Lebesgue:
\begin{equation*}
m^*(A)=inf \lbrace \sum_{n=1}^\infty (b_n-a_an): a_n \leq b_n \in \mathbb{R}, A\subset \bigcup_{n=1}^\infty (a_n,b_n]
\end{equation*}

Qué es una medida exterior. Por el teorema de Caratheodory, sabemos que si restringimos la medida exterior a la $\sigma$-álgebra de los medibles, tenemos una medida, y además al estar generada por la medida $\rho$ del semianillo de los intervalos semiabiertos, estos mismos son conjuntos medibles.

Ya que los conjuntos abiertos se pueden expresar cómo uniones numerables de intervalos semi-abiertos, se tiene que la medida de Lebesgue mide a los abiertos, por lo tanto mide a la menor $\sigma$-álgebra que los contiene, los Borelianos.

\end{ej}

\section{Construcción de la medida de Hausdorff}




\begin{defi}
\rm
 Sea un espacio métrico (X,d), A$\subset X$, el diámetro de A se define cómo:
\begin{equation*}
diam(A)=\sup_{x,y\in A}\{d(x, y): x,y\in A \}
\end{equation*}
\end{defi}

Para $\delta > 0$ y $p >0$ definamos una función de conjunto  $H_{\delta}^{T}:\mathcal{P}(X)\longrightarrow \mathbb{R}$ , que tiene cómo valor:
\\\\
\begin{equation}
H_{\delta}^{T}(A)=inf\{\sum_{i\in\mathbb{N}}diam(U_i)^t: A \subset \bigcup_{i \in \mathbb{N}}U_i, diam(U_i)<\delta \}
\end{equation}
\\\\
Observemos que la única restricción que tienen los conjuntos que cubren a A es el diámetro.\\

Gracias a la proposición 1, las anteriores funciones son medidas exteriores.Observemos que cuando $\delta < \epsilon$:
\begin{equation*}
inf\{\sum_{i\in\mathbb{N}}diam(U_i)^t: A \subset \bigcup_{i \in \mathbb{N}}U_i, diam(U_i)<\delta \}  \geq inf\{\sum_{i\in\mathbb{N}}diam(U_i)^t: A \subset \bigcup_{i \in \mathbb{N}}U_i, diam(U_i)<\epsilon \} 
\end{equation*}
\begin{equation}\label{epsilonDelta}
 H_{\delta}^t(A)>H_{\epsilon}^t(A)
\end{equation}
Observemos que, $\delta<\epsilon$, la familia de conjuntos con diámetro menor que $\delta$, está contenida en la familia de conjuntos con radio menor que $\epsilon$, por lo tanto al ser una familia más grande el ínfimo será menor para, la familia de conjuntos con radio menor que $\epsilon$
\begin{defi}
\rm
 Llamamos medida exterior t-dimensional de Hausdorff a:

\begin{equation}
H^t=\lim_{\delta\rightarrow0}H_{\delta}^t
\end{equation}
\end{defi}
\begin{prop}
 La medida exterior t-dimendional de Hausdorff es una medida exterior

\end{prop}

\begin{proof}
Comprobemos que $H^t$ cumple con la definición de medida exterior, partiendo de que $H_\delta^t$ lo es:
\begin{itemize}
\item \[\hspace{1cm} H^t(\emptyset)=\lim_{\delta\rightarrow0}H_{\delta}^t(\emptyset)=\lim_{\delta\rightarrow0}0=0\]
\item \[ \hspace{1cm} A \subseteq B\Rightarrow H^t_\delta(A)\leq H^t_\delta(B).\]Entonces, por la regla del sandwich sabemos que:
\[ \lim_{\delta\rightarrow0} H^t_\delta(A)\leq H^t_\delta(B)\Rightarrow H^t(A) \leq H^(B) \] 

\item  Subaditividad, sabemos que:  
\[ H^t(\bigcup_{i \in \mathbb{N}}A_i)=\lim_{\delta\longrightarrow 0} H^t_\delta(\bigcup_{i \in \mathbb{N}}A_i)\leq \lim_{\delta\longrightarrow 0} \sum_{i \in \mathbb{N}}H^t(A_i), \]por \ref{epsilonDelta} y sabeindo que $0\leq \delta$ se tiene: 
\[  \sum_{i \in \mathbb{N}}H_\delta^t(A_i) \leq \sum_{i \in \mathbb{N}}H^t(A_i) \] Por lo tanto llegamos a la conclusión de:
\[ H^t(\bigcup_{i \in \mathbb{N}}A_i) \leq sum_{i \in \mathbb{N}}H^t(A_i) \]
\end{itemize}







\end{proof}






De a cuerdo con el teorema de Caratheodory, sabemos que si restringimos la medida exterior de Hausdorff a la familia de sus conjuntos medibles obtenemos una medida, veamos ahora que los borelianos están contenidos en $M_{H_p}$



\begin{defi}
\rm
 Sea (X, d) un espacio métrico, diremos qué una medida exterior $\mu^*$ es métrica, si dades A,B$\subset$X tales que d(A,B)$\ge$0, la medida exterior de la unión es la suma de medidas exteriores
\end{defi}

\begin{prop}[pg 40 UNEX]
 Si $\mu^*$ es una medida exterior métrica en un espacio métrico (X,d), entonces:
\begin{equation*}
\mathcal{B}(X) \subset M_{\mu^*}
\end{equation*}
\end{prop}
\begin{proof}
Basta demostrar que todos los abiertos en X son conjuntos $\mu^*$-medibles, es decir si U es abierto, $\forall A \subseteq X$:
\begin{equation*}
\mu^*(A)=\mu^*(A\cap U)+\mu^*(A\cap U^c)
\end{equation*}
Sea A t.q. $\mu^*(A)<\infty$, probaremos qué: \[ \mu^*(A)\geq \mu^*(A\cap U)+\mu^*(A\cap U^c)\], ya que, $ \mu^*(A)\leq \mu^*(A\cap U)+\mu^*(A\cap U^c$ es una propiedad de la medida exterior, y el caso para infinito es trivial.
\\\\
Definamos la familia de conjuntos: \[U_n = \lbrace x \in X : d(x, U^c) \geq\frac{1}{n}\rbrace\] Notemos qué al ser $U_n$ abierto, $U_n\uparrow U$. Y por lo tanto, $A\cap U_n\uparrow A \cap U$  y: \[d(A\cap U_n, A \cap U^c)\geq d(U_n, U^c) \geq \frac{1}{n}\Rightarrow A\supseteq (A\cap U_n)\cup (A \cap U^c)\]De la definición de medida exterior y de medida exterior métrica deducimos que: 
\[ \forall n,\hspace{0.2cm} \mu^*(A)\geq \mu^*((A\cap U_n)\cup (A \cap U^c))= \mu^*(A\cap U_n) +\mu^* (A \cap U^c)\]
Si consideramos la familia de conjuntos \[B_n=A\cap U_n \uparrow B=A\cap U .\] Basta demostrar que $\mu^*(B_n)\uparrow\mu^*(B)$.
\\\\ Sea la familia de conjuntos disjuntos \[C_1=B_1, n\geq2, C_n=B_n-B_{n-1}\]
\[B=B_n\cup \bigcup_{i=n}^{ \infty }\lbrace C_i \rbrace \Rightarrow \mu^*(B)=\mu^*(B_n\cup \bigcup_{i=n}^{\infty}\lbrace C_i \rbrace)\leq \mu^*(B_n) + \sum_{i=n}^{\infty}(C_i)\]
\[\Rightarrow \mu^*(B)-\mu^*(B_n) \leqslant \sum_{i=n}^{\infty}(C_i)\]
Por lo tanto ahora tenemos que ver qué: $\sum_{i=n}^{\infty}(C_i) \longmapsto 0$, así qué basta demostrar qué: $\sum_{i=0}^{\infty}(C_i)$ converge,\\ Para ello veamos que si \[x \in B_{n-1} \hspace{0.2cm}\wedge \hspace{0.2cm} y \in C_{n+1}\Rightarrow d(x,y)\geq d(B_{n-1}, C_{n+1})\] \\ ya que sabemos que: \[d(B_{n-1}, U^c) \geq \frac{1}{n-1}\hspace{0.2cm} \wedge \hspace{0.2cm} d(C_{n+1}, U^c) \geq \frac{1}{n}\]
Por lo tanto: \[d(B_{n-1}, C_{n+1}) \geq \frac{1}{n-1}-\frac{1}{n}=\frac{1}{n(n-1)}>0\]
Ahora de las definiciones de medida exterior y de medida exterior métrica:

\[ \mu^*(B_{n-1})+\mu^*(C_{n+1})=\mu^*(B_{n-1}\cup C_{n+1}) \leq \mu^*(B_{n+1})\] 


De modo que:\\$$\mu^*(C_{n+1}) \leq \mu^*(B_{n+1})-\mu^*(B_{n-1}), $$
$$\mu^*(C_{1})+\mu^*(C_{3})+...+\mu^*(C_{2n-1})\leq\mu^*(B_{1})+\mu^*(B_{3})-\mu^*(B_{1})+..=\mu^*(B_{2n-1}),$$

(Notar que a $B_{n-1}\cup C_{n+1}$ le falta $C_n$ para cubrir a $B_{n+1}$)
\\Haciendo lo propio para los pares e impares:\\

\[ \sum_{i=1}^n \mu^*(C_{2i-1}) \leq \mu^*(B_{2n-1})\leq \mu^*(A)\]
\\
\[ \sum_{i=1}^n \mu^*(C_{2i}) \leq \mu^*(B_{2n})\leq \mu^*(A)\]
\\ Y cómo por hipótesis $\mu^*(A)\leq \infty$ la serie $\sum\mu^*(C_n)$ converge.

\end{proof}


\begin{teo}
$H_p:\mathcal{P}(X)\longrightarrow [0, \infty] $ es una medida exterior métrica
\end{teo}

\begin{proof}

Sean A, B t.q. d(A,B)$>$0. Ahora sea $d(A,B)>\delta>0$.
\\Sea una sucesión de conjuntos 
\[\{C_n\}_{n \in \mathbb{N}}: \hspace{0.2cm}diam(C_n)<\delta,\hspace{0.2cm} A\cup B \subset\cup C_n\]
Ahora notar que al tomar $\delta$ lo suficientemente pequeño es imposible que un $C_n$ interseque al mismo tiempo con A y B, ya que la distancia entre A y B es mayor que el diámetro de estos conjuntos, teniendo este concepto en cuenta definamos dos familias de índices disjuntas:
\[I=\{i\in \mathbb{N}:C_n\cap A \neq \emptyset\}\hspace{1cm}J=\{j\in \mathbb{N}:C_n\cap B \neq \emptyset\}\]Se cumple que $A\subseteq\cup_{i\in I}C_i$ y $B\subseteq\cup_{j\in J}C_i$. Por la definición de medida exterior:
\[H_\delta^t(A\cup B)\leqslant H_\delta^t(A)+H_\delta^t(B)\] Y ya que $H_\delta^t$, está definida sobre el ínfimo de los cubrimientos y hemos tomado uno cualquiera: \[ H_\delta^t(A)+H_\delta^t(B)\leqslant \sum_{i\in I}(diam(C_i))^t)+\sum_{j\in J}(diam(C_j))^t)\leqslant \sum_{i\in \mathbb{N}}(diam(C_i))^t)\] Ahora tomando ínfimo \[ H_\delta^t(A)+H_\delta^t(B)\leqslant  inf\sum_{i\in \mathbb{N}}(diam(C_i))^t)=H_\delta^t(A\cup B)\] Y por el lema del sandwich concluimos que: \[H_\delta^t(A\cup B)= H_\delta^t(A)+H_\delta^t(B)\] 
\end{proof}


Por lo tanto ya hemos llegado a la conclusión de que la medida exterior de Hausdorff restringida a sus medibles es una medida positiva y que los Borelianos son conjuntos $H_p$-medibles.

\section{Caracterización de la dimensión de Hausdorff}

El siguiente resultado nos dará pie a poder definir la dimensión de Hausdorff, así como el poder asegurar su existencia.
\begin{teo}
Sea $A\subset X$ y sean 0$\leqslant p<q$, entonces se tiene que:
\begin{itemize}
	\item $$Si \quad H_p(A)<\infty \quad \Rightarrow \quad H_q(A)=0, $$
	\item $$Si \quad H_q(A)>0 \quad \Rightarrow \quad H_q(A)=\infty. $$
\end{itemize}

\end{teo}

\begin{proof}
Basta con probar solo la primera sentencia, ya que la segunda es la contrarecíproca de la primera, es decir si la primera fuera ($A \Rightarrow B$), se tiene que:
$$(A \Rightarrow B) \Leftrightarrow (\overline{B}\Rightarrow \overline{A})$$
Ya que la medida de Hausdorff es una función positiva, se tiene que negar que $H_q(A)=0$, es decir que $H_q(A)>0$, y negar que $H_p(A)<\infty$, es decir que $H_p(A)=\infty$.\\
Ahora sea $\delta >0$ y $B_n$ un cubrimiento de A, con d($B_n)<\delta$, se tiene que:
$$H_{\delta}^q(A)\leq \sum d(B_n)^q, $$
y ya que:
$$d(B_n)^q=d(B_n)^pd(B_n)^{q-p}\leqslant d(B_n)^p\delta^{q-p},$$
se tiene que:

$$H_{\delta}^q(A)\leq \sum d(B_n)^q<\delta^{q-p} \sum d(B_n)^p.$$

Ahora tomando el ínfimo, se tiene que, $H_{\delta}^q(A)\leqslant \delta^{q-p}H_{\delta}^p(A)$, ahora si hacemos $\delta$ tender a 0, $H^p(A)<\infty.$
\end{proof}
\begin{defi}
De este teorema deducimos que existe un valor crítico $s_0\in [0,\,\, ,\infty]$, tal que :

\begin{itemize}
	\item $$\forall s<s_0, \quad H^s(A)=\infty, $$
	\item $$\forall s>s_0, \quad H^s(A)=0. $$
\end{itemize}
A este valor $s_0,$ le llamaremos la dimensión de Hausdorff del conjunto A. Denotaremos dim(A)=$s_0$.
\end{defi}

\begin{nota}
Si para todo s mayor que 0, $H^s(A)=0$, entonces dim(A)=0. Y si para todo s mayor que 0 también, $H^s(A)=\infty$, entonces dim(A)=$\infty$.

\end{nota}

\subsection{Teorema del contenido}\label{Teorema del contenido}
\begin{teo}[Teorema del contenido]
Sean A, B conjuntos Borelianos, entonces si $A \subseteq B$, se tiene que:

$$dim(A)\leqslant dim(B). $$
\end{teo}

\begin{proof}
Supongamos que, $A \subseteq B$. Veamos entonces que $dim(A)\leqslant dim(B).$
\\La dimensión de Hausdorff de B es ese valor $s_0$, tal que $\forall s>s_0$, la medida de Hausdorff de B es 0, sea $s>s_0$, entonces:
$$ H^s(A)\leqslant H^s(B)=0,$$
Por lo tanto diam(A)$\leqslant$dim(B).
\end{proof}
\subsection{Teorema de la unión}
\begin{teo}[Teorema de la unión]
Sean A, B conjuntos Borelianos, entonces $$ dim(A\cup B)=max\lbrace dim(A),\,\,dim(B)\rbrace.$$
\end{teo}

\begin{proof}
Por el teorema \ref{Teorema del contenido} sabemos que dim($A\cup B$)$\geqslant$max$\lbrace dim(A), dim(B)\rbrace$. Por lo tanto hay que probar que dim($A\cup B$)$\leqslant$max$\lbrace dim(A), dim(B)\rbrace$.\\
Sea $s>max\lbrace dim(A),\,\,dim(B) \rbrace$, por lo tanto $H^s(A)=0$ y $H^s(B)=0$. Entonces \textcolor{red}{$$H^s(A\cup B)\leqslant H^s(A)+H^s(B)=0.$$ } Por lo tanto $\forall s >$max$\lbrace dim(A), dim(B) \rbrace$, dim($A\cup B$)$\leqslant$ s. Así que: $$dim(A\cup B)\leqslant max\lbrace dim(A), dim(B) \rbrace.$$
\end{proof}
\chapter{Autosemejanza}

\begin{defi}
\rm
 
Una semejanza (o similitud) es una aplicación entre dos espacios métricos,($X_1, d_1$) y ($X_2, d_2$) que modifica las distancias entre dos puntos cualesquiera multiplicándolas por un factor fijo, r. En el caso de los espacios euclídeos, por ejemplo, es la composición de una isometría y una homotecia. Intuitivamente, es una transformación que puede cambiar el tamaño y la orientación de una figura pero no altera su forma.
\end{defi}
Una lista de radios, es una lista finita de números positivos ($r_1, r_2, ...,r_n $).
Si $\forall i$ $r_i$<1 la lista de radios se dice contractiva o hiperbólica
\begin{defi}
\rm
 
En un espacio métrico (X,d), un sistema de funciones iterado, relativa a la lista de radios($r_1, r_2, ..,r_n$), es una lista ($f_1, f_2,...,f_n$), con $f_i$ una semejanza de radio $r_i>0$
\end{defi}
Sea $K\subset X$ se dice que es invariante o atractor por el sistema de funciones iterado ($f_1, f_2,...,f_n$) sii $$K=\bigcup_{i=1}^nf_i[k]$$
\begin{defi}
\rm

El valor de semejanza de una lista de radios ($r_1, r_2, ..,r_n$), es un número positivo s tal que:
$$\sum_{i=1}^nr_i^s=1$$
\end{defi}
\begin{teo}[pg117Edgar]

Si ($r_1, r_2, ..,r_n$) es una lista de radios contractiva, entonces existe un único valor de semejanza s. Se tiene que s=0 sii n=1.
\end{teo}
Al número s, se le llama dimensión de semejanza de un conjunto K que sea invariante por un conjunto iterado de funciones relativa a la lista de radios ($r_1, r_2, ..,r_n$), es decir:
\\
\\
Sea la lista de radios ($r_1, r_2, ..,r_n$) tal que, $$\sum_{i=1}^nr_i^s=1,$$ con la lista iterada de funciones relativa a ella ($f_1, f_2,...,f_n$), cada $f_i$ es una semejanza con radio $r_i$, entonces si el conjunto K es invariante por ($f_1, f_2,...,f_n$), es decir: $$K=\bigcup_{i=1}^nf_i[k],$$ se dice que K tiene dimensión de semejanza s.
\\\\
Intuitivamente la dimensión de semejanza, debería coincidir con la de Hausdorff y esta es única pero sin las caracterizaciones adecuadas el conjunto K puede admitir dos dimensiones de semejanza, así que esta mas que una característica del conjunto K es una característica del conjunto iterado de funciones ($f_1, f_2,...,f_n$).

\begin{ej}
Veamos un ejemplo para comprobar que efectivamente un conjunto puede admitir más de una dimensión de semejanza. \\
Sea el intervalo [0,1], este conjunto es invariante por las semejanzas: $f_1(x)=\frac{x}{2}$ y $f_2(x)=\frac{x}{2}+\frac{1}{2}$, que tienen radio $\frac{1}{2}$ cada una, ya que $$2*\frac{1}{2}^s=1\Leftrightarrow s=1,$$ se tiene que según esta lista de funciones la dimensión de semejanza de [0,1] es 1.\\
En cambio si ahora tomamos las semejanzas $f_3(x)=\frac{2x}{3}$ y $f_4(x)=\frac{2x}{3}+\frac{1}{3}$, igualmente [0,1] es invariante por la lista de funciones, pero en este caso la dimensión de semejanza sale mayor que 1.
\end{ej}
De este ejemplo podemos intuir que habrá que evitar el solapamiento de los conjuntos imagen.
\begin{defi}

\section{Orden de una familia de conjuntos}

\begin{defi}
\rm

Para un $n \in \mathbb{Z}$, con $n\geq -1$, el orden de una familia de conjuntos distintos $\mathcal{A}$ será menor o igual que n si cualquiera n+2 conjuntos de la familia tienen intersección vacía.\\
Se dice que $\mathcal{A}$ tiene orden exactamente $n\geqslant 0$ si tiene orden menor o igual que n pero no menor o igual que n-1.
\end{defi}

\begin{nota}
Se dice que una familia de conjuntos $\mathcal{A}$ tiene orden exáctamente -1 si tiene orden $n\geqslant -1$ 
\end{nota}

\begin{nota}
Una familia de conjuntos $\mathcal{A}$ tendrá orden $\infty$, si $\forall n \, \in \mathbb{N}$, existen n conjuntos distintos pertenecientes a la familia $\mathcal{A}$, con intersección distinta del vacío. Es decir:

$$\forall n \in \mathbb{N},\tab \exists A_i \in \mathcal{A}, \,\,\, i=1,...,n:\, \bigcap_{i=1}^n A_i \neq \emptyset  $$ 

\end{nota}

\end{figure}

\section{Espacios de dimensión cero}
Esta sección estará dedicada a entender un poco mejor y caracterizar brevemente los espacios de dimensión cero.

Antes de nada, será indispensable para el desarrollo de la teoría que sigue, conocer el concepto de cubrimiento.
\begin{defi}
\rm
Se dice que una familia de conjuntos $\mathcal{A}$, es un cubrimiento del espacio métrico S, que $\mathcal{A}$ cubre a S, si y solo si S está contenido en la unión de todos los elementos de $\mathcal{A}$.

\end{defi}

\begin{defi}
\rm

Sean $\mathcal{A}$ y $\mathcal{B}$ dos colecciones de conjuntos, entonces se dice que $\mathcal{B}$ está subordinado a $\mathcal{A}$ si y solo si:
$$\forall B \in \mathcal{B}, \quad \exists A \in \mathcal{A}: \quad B\subseteq A.$$
Si $\mathcal{A}$ es un cubrimiento abierto del espacio métrico S, un refinamiento de $\mathcal{A}$ es un cubrimiento abierto $\mathcal{B}$ de S que es subordinado a $\mathcal{A}$.
\end{defi}

\begin{defi}
\rm

Sea un espacio métrico S. $A \subseteq S$ es un conjunto clopen, si es cerrado y abierto al mismo tiempo. En particular S y $\emptyset$ son conjuntos clopen.

\end{defi}
\begin{defi}
\rm


Sea el espacio métrico S. Una partición clopen de S se le dice a una familia de conjuntos en cuya unión está contenido S, son disjuntos dos a dos (partición) y cada uno de ellos es clopen.
\end{defi}

\begin{defi}
\rm

Un espacio S es de dimensión cero si y solo si todo cubrimiento abierto y finito de S tiene un refinamiento que es una partición clopen.
\end{defi}

En la siguiente sección veremos la relación que existe entre los conceptos de orden de una familia de conjuntos y de dimensión topológica (dimensión de cubrimientos).

\section{Dimensión de cubrimientos}


Sea S un espacio métrico, y n un número entero $\geq -1$. Se dice que S tiene dimensión de cubrimiento menor o igual que n si y solo si todo cubrimiento con cardinal finito de S posee un refinamiento de orden menor o igual que n.\\
La dimensión de cubrimiento de S será exactamente n, para $n\geqslant 0$ si y solo si es menor o igual que n pero no menor o igual que n-1. Y se escribe Cov(S)=n.\\ Esta también llamada dimensión de Lebesgue.\\
\\
\begin{nota}
Al igual que pasaba con el orden de un conjunto, la dimendión de cubrimientos de un conjunto diremos que es exactamente igual a -1 si y solo si es menor o igual que -1.

\end{nota}
De las definiciones de orden de un conjunto y dimensión de cubrimientos, podemos deducir que:

$$Cov(S)=-1\Longleftrightarrow S=\emptyset \vee S=\lbrace \emptyset \rbrace,$$

$Cov(S)=0\Longleftrightarrow$ S es un espacio de dimensión cero ya que se puede cubrir por una partición de conjuntos disjuntos dos a dos, que además estos conjuntos serán clopen en S.\\\\
A continuación los siguientes resultados nos servirán como herramienta para probar los teoremas de suma y contenido de dimensión de cubrimientos.
\begin{nota}\label{topologiaElemental}

Sean F cerrado, U un abierto y $F\subseteq U$ entonces: $$\exists V: F\subseteq V \subseteq \overline{V} \subseteq U,$$ con V abierto
\end{nota}
Este resutado,  nota \ref{topologiaElemental}, fué visto en la asignatura de topología por lo que prenscindiremos de demostración.
\begin{teo}\label{teo}

Sea S un espacio métrico y n un número entero $\geq -1$, las siguientes afirmaciones son equivalentes:
\begin{enumerate}
  \item $Cov(S\leq n)$
  \item Si $\mathcal{U}=\lbrace U_1, U_2,..,U_k \rbrace$ es un cubrimiento abierto y finito de S entonces, existe un cubrimiento $\mathcal{B}=\lbrace B_1, B_2,..,B_k \rbrace$ que refina a $\mathcal{U}$, tal que $U_i\supseteq B_i,\,\, i=1,...,k$, tiene su mismo número de elementos y cubre a S con orden menor o igual que n.
  \item Si $\mathcal{U}=\lbrace U_1, U_2,..,U_{n+2} \rbrace$ es un cubrimiento abierto y finito de S entonces, existe un refinamiento de $\mathcal{U}$, $\mathcal{B}=\lbrace B_1, B_2,..,B_{n+2} \rbrace$, tal que $U_i\supseteq B_i,\,\, i=1,...,n+2$, que también tiene n+2 elementos, sigue cubriendo a S y: $$\bigcap_{i=1}^{n+2}B_i=\emptyset.$$
  \item  Si $\mathcal{U}=\lbrace U_1, U_2,..,U_{n+2} \rbrace$ es un cubrimiento abierto de S entonces, existen conjuntos cerrados  $\lbrace F_i\subseteq U_i\rbrace_{i=1}^{n+2}$ que sigue cubriendo a S y: $$\bigcap_{i=1}^{n+2}F_i=\emptyset.$$
  \end{enumerate}
 
\end{teo}

\begin{proof}
(2)$\Rightarrow$(3) es claro ya que (2) es una generalización de (3) y (2)$\Rightarrow$(1) también por definición de dimensión de cubrimiento.

\begin{enumerate}
	\item(1)$\Rightarrow$(2)\\
	
	Supongamos que Cov(S)$\leq$n. Sea un cubrimiento abierto de S, $\mathcal{U}=\lbrace U_1,...,U_k \rbrace$.\\ Por definición de dimensión de cubrimientos, $\mathcal{U}$ posee un refinamiento al que llamaremos $\mathcal{W}$ que tiene orden $\leq$ n, ahora para probar (2) debemos ser capaces de construir uno, que lo haremos a partir de $\mathcal{W}$ y con  exáctamente k elementos.\\ Fijémonos en que: $$\forall W \in \mathcal{W}, \quad \exists U_i\in \mathcal{U}:\,\, W\subseteq U$$.\\Definamos la familia de conjuntos $\mathcal{B}$ de la siguiente forma : $$B_i=\bigcup \lbrace W\in \mathcal{W}:\quad W\subseteq U_i  \rbrace ,$$ ya que $\mathcal{W}$ cubría ya a S, los $B_i\subseteq U_i$ por como están construidos también cubren a S, además hemos restringido su cardinal a k, ahora, ¿siguen teniendo orden $\leq$ n?\\Supongamos que x$\in$S, x como mucho puede estar simultáneamente en n+1 elementos de $\mathcal{W}$ y x $\in B_i$ solo si x$\in W$ para algún W$\subseteq U_i$ por lo tanto x pertenece como mucho a n+1 $B_i$'s, lo que quiere decir que el orden de  $\mathcal{B}$ es n. 
	\item (3)$\Rightarrow$(2)\\ 
	Supongamos que se cumple (3). Sea el cubrimiento abierto de S, $\lbrace U_1, U_2,..,U_k \rbrace$, si k$\leq n+1$ entonces el cubrimiento ya tendría orden $\leq n$ y no habría nada que probar.\\ Supongamos entonces que k$\geq n+2$. Sea la familia de conjuntos $\lbrace W_i\rbrace_{i=1}^{n+2}$, definida como, $W_i=U_i$, i=1,...,n+1 y $W_{n+2}=\bigcup_{i=n+2}^k U_i $. Ya que los $U_i$'s cubren a S, los $W_i$'s también lo cubren.\\ Por (3): $$\forall i \quad \exists V_i: \quad V_i\subseteq W_i, \tab \bigcap_{i=1}^{n+2}V_i=\emptyset,  \tab \bigcup_{i=1}^{n+2}V_i=S,$$ para probar (2) solo nos queda ajustar el número de elementos de n+2 a k, para ello definiremos los $B_i$'s tales que: $$ B_i=V_i,\tab i=1,..,n+1, \tab B_j=V_{n+2}\cap U_j,\tab j\geq n+2,$$ entonces por construcción los $B_i$'s cumplen que: $$ \bigcap_{i=1}^{k}B_i=\emptyset, \tab \bigcup_{i=1}^{k}B_i=S\tab y \tab \forall i, B_i\subseteq U_i. $$
	\item $(3)\Longrightarrow (4)$\\\\
	
Supongamos cierto (3), es decir, dado el cubrimiento abierto de S: 

$$\lbrace U_1, U_2,...,U_{n+2}\rbrace, \quad \forall i=1,...,n+2, \quad \exists B_i \subseteq U_i:$$ $$ \bigcap_{i=1}^{n+2}B_i=\emptyset \quad y \quad \bigcup_{i=1}^{n+2}B_i=S, $$	
	Con cada $B_i$ conjunto abierto en S. \\ Dado que la unión de todos los $B_i$'s es S se tiene que:
	
	$$S\smallsetminus B_1 \subseteq \bigcup_{i=2}^{n+2}B_i, $$
	
	Ahora usando la nota \ref{topologiaElemental} y que la unión finita de conjuntos abiertos es a su vez un conjunto abierto, deducimos que $\exists \,\, V_1$, tal que:
	
$$S\smallsetminus B_1 \subseteq V_1 \subseteq \overline{V_1} \subseteq \bigcup_{i=2}^{n+2}B_i. $$

Si definimos el conjunto $F_1=S\smallsetminus V_1,$ podemos apreciar que cumple con las siguientes propiedades:

$$S\smallsetminus B_1 \subseteq V_1 \Longrightarrow S\smallsetminus V_1=F_1\subseteq B_1 \Longrightarrow F_1\cap \bigcap_{i=2}^{n+2}B_i=\emptyset \quad y $$	

$$V_1 \subseteq \bigcup_{i=2}^{n+2}B_i  \Longrightarrow S\smallsetminus V_1=F_1 \supseteq S\smallsetminus  \bigcup_{i=2}^{n+2}B_i\Longrightarrow F_1 \cup \bigcup_{i=2}^{n+2}B_i=S.$$

Siguiendo el mismo procedimiento existe un conjunto abierto $V_2$, tal que:

$$S\smallsetminus B_2 \subseteq V_2 \subseteq \overline{V_2} \subseteq (S\smallsetminus \overline{V_1})\cup \bigcup_{i=3}^{n+2}B_i. $$
Ahora si $F_2=S\smallsetminus V_2$, se tiene que:

$$F_1\cup F_2 \cup \bigcup_{i=3}^{n+2}B_i=S\quad y \quad F_1\cap F_2 \cap \bigcap_{i=3}^{n+2}B_i=\emptyset ,$$

Continuando con este procedimiento acabaremos definiendo todos los $F_i$'s.
	
	\item (4)$\Rightarrow$(3)\\ 
	Suponemos cierto (4).Tenemos que: $$F_1\subseteq (U_1\cap(S\smallsetminus \bigcap_{i=2}^{n+2}F_i)),$$ Esto es cierto ya que $F_1\subseteq U_1$, por la propia definición de los $F_i's$ y que $F_1\subseteq S/\bigcap_{i=2}^{n+2}F_i)$, también es cierto porque $\bigcap_{i=1}^{n+2}F_i=\emptyset$.\\
Usando el resultado correspondiente a la nota \ref{topologiaElemental}, existe un conjunto abierto $B_1$ tal que:$$F_1\subseteq B_1 \subseteq \overline{B_1} \subseteq ( U_1\cap(S \smallsetminus\bigcap_{i=2}^{n+2}F_i)),$$ Por como está definido $B_1$, se tiene que $\overline{B_1}\subseteq U_1$ y que $\overline{B_1}\cap \bigcap_{i=2}^{n+2}F_i=\emptyset$.\\Luego también existe un conjunto abierto $B_2$, tal que, 
$$F_2\subseteq B_2 \subseteq \overline{B_2} \subseteq ( U_2\cap(S\smallsetminus(\overline{B_1}\cap\bigcap_{i=3}^{n+2}F_i))),$$
por lo tanto, $\overline{B_2}\subseteq U_2$ y $\overline{B_1}\cap \overline{B_2}\cap \bigcap_{i=3}^{n+2}F_i=\emptyset$, continuando la demostración de esta forma al final llegamos a que existe una familia de conjuntos abiertos, $B_i$'s i=1,...,n+2, tales que:
$$B_i\subseteq U_i, \quad S\subseteq \bigcup_{i=1}^{n+2}B_i\quad y \quad \bigcap_{i=1}^{n+2}B_i=\emptyset. $$
	
\end{enumerate}
\end{proof}



Se denomina ancho de red de un cubrimiento al mayor de los diámetros de los conjuntos que lo forman.
\begin{prop}\label{propAbiertos}

Sea S un espacio métrico con $Cov(S)\leq 1$. Tomemos A, B, C subconjuntos cerrados de S tales que, A$\cap$B$\cap$C$=\emptyset$, entonces existen conjuntos abiertos U,V,W cumpliendo que:\\$$U\supseteq A, V\supseteq B, W\supseteq C, \tab U\cap W\cap W=\emptyset \tab U\cup V \cup W=S.$$
\end{prop}

\begin{proof}

Si A,B,C son cerrados, entonces sus complementarios A', B' y C' son abiertos y además, aplicando las Leyes De Morgan: $$ (A\cap B\cap C)'=\emptyset '\Rightarrow A'\cup B'\cup C=S,$$ cubren a S, A'$\cup$B'$\cup$C'= S, entonces por el teorema \ref{teo}, existen conjuntos cerrados F,G,H,  F$\subseteq$A', G$\subseteq$B' y H$\subseteq$C', que cubren a S y tienen intersección vacía. Ahora sus complementarios F',G',H', son conjuntos abiertos en S tales que, por coonstrucción: A$\subseteq$F', B$\subseteq$G' y C$\subseteq$H', $F' \cap G' \cap H'= \emptyset$ y $F' \cup G' \cup H'= S$ 
\end{proof}



\begin{prop}\label{PropCerrados}

Sea S un espacio métrico con $Cov(S)\leq 1$. Tomemos A, B, C subconjuntos cerrados de S tales que, A$\cap$B$\cap$C$=\emptyset$, entonces existen conjuntos cerrados U,V,W cumpliendo que:\\U$\supseteq$A, V$\supseteq$B, W$\supseteq$C, U$\cap$V$\cap$W$=\emptyset$ y U$\cup$V$\cup$W$=S$
\end{prop}
\begin{proof}

Usando la proposición \ref{propAbiertos} y 4 en el teorema \ref{teo}, la demostración es trivial.
\end{proof}

\begin{prop}\label{2.2.7Edgar}

Sean $A_1, A_2, A_3$ subconjuntos cerrados de S tales que $A_1\cap A_2\cap A_3= \emptyset$. Entonces existen $U_1, U_2, U_3$ abiertos tales que: $$i=1,2,3, \tab U_i\supseteq A_i, \tab \overline{U_1}\cap \overline{U_2} \cap \overline{U_3}= \emptyset.  $$
\end{prop}


\begin{proof}
Para i=1,2,3, definamos las funciones $d_i=d(x, A_i)$. Ya que $A_1\cap A_2\cap A_3= \emptyset$, $\forall x \in S, \tab d_1(x)+d_2(x)+d_3(x)>0$. Ahora para i=1,2,3, definamos los conjuntos: $$U_i=\lbrace x \in S: d_i(x)<\frac{1}{4}(d_1(x)+d_2(x)+d_3(x)) \rbrace. $$ Los $U_i$'s son abiertos y además $U_i\supset A_i$, ya que si $x \in A_i \Rightarrow d_i(x)=0.$\\Ahora ¿$\overline{U_1}\cap \overline{U_2}\cap \overline{U_3}=\emptyset$? Supongamos que $x\in \overline{U_1}\cap \overline{U_2}\cap \overline{U_3}$, entonces:$$ i=1,2,3,\tab d_i(x)<\frac{1}{4}(d_1(x)+d_2(x)+d_3(x)),$$Si $  d_1(x)<\frac{1}{4}(d_1(x)+d_2(x)+d_3(x))\Rightarrow d_1(x)<\frac{d_2(x)+d_3(x)}{3}.$\\
$d_2(x)<\frac{1}{4}(d_1(x)+d_2(x)+d_3(x))\Rightarrow d_2(x)<\frac{d_3(x)}{2}$ y como también $d_3(x)<\frac{1}{4}(d_1(x)+d_2(x)+d_3(x))$ llegamos a la conclusión que, $  d_3(x)<\frac{d_2(x)}{2} $, lo cual es absurdo ya que contradice  lo anterior. 
\end{proof}




\subsection{Teoremas de suma}

\begin{teo}\label{3.2.9 Edgar}

Sea S un espacio métrico, sean $F_1$ y $F_2$ subconjuntos cerrados de S tales que $Cov(F_1)\leq 1$ y $Cov(F_2)\leq 1$. Entonces $Cov(F_1\cup F_2)\leq 1$
\end{teo}

\begin{proof}
Para probar el teorema debemos ver que todo cubrimiento de $F_1\cup F_2$ posee un refinamiento  de orden $\leq 1$ Sean A, B, C, conjuntos cerrados en $F_1 \cup F_2$ tales que $A \cap B \cap C= \emptyset$. Sus intersecciones con $F_1$, A$\cap F_1$, B$\cap F_1$, C$\cap F_1$, son cerrados en $ F_1$, y con intersección vacía, $A \cap B \cap C \cap F_1= \emptyset$, entonces por (Proposición \ref{PropCerrados}), existen conjuntos cerrados K, L, M que contienen a los anteriores, K$\supseteq (A\cap F_1)$, L$\supseteq (B\cap F_1)$, M$\supseteq (C\cap F_1)$, cubren a $F_1$, $F_1= K \cup L \cup M$ y su intersección es vacía $K \cap L \cap M = \emptyset$. \\ Podemos definir los conjuntos:$$(K\cup A)\cap F_2, (L\cup B)\cap F_2, (M\cup C)\cap F_2, $$ que por construcción sabemos que también tienen intersección vacía y son cerrados en $F_2$ y están contenidos en $F_2$, por el mismo argumento de antes transladado a $F_2$ existen conjuntos cerrados P, Q, R, tales que: $(K\cup A)\cap F_2\subseteq P$, $(L\cup B)\cap F_2 \subseteq Q$,$(M\cup C)\cap F_2\subseteq R$, $P\cap Q\cap R = \emptyset$ y $P\cup Q\cup R=F_2$. Ya estamos en condiciones de construir un cubrimiento de $F_1 \cup F_2$ con orden menor o igual que 1, este es $E=P\cup K$, $F=Q \cup L$ y $G=R\cup M$ que por como están construidos estos conjuntos sabemos que : \begin{enumerate}
	\item E, F, G son cerrados.	
	\item $E\cap F\cap G= \emptyset,$
	\item $E\cup F\cup G = F_2, $
\end{enumerate} 
ya hemos econtrado para un cubrimiento cualquiera de $F_1\cup F_2$ tiene un refinamiento con orden menor o igual que 1. Por lo que acabamos de promar que Cov($F_1\cup F_2$)$\leq 1$ 
\end{proof}

\end{proof}


\begin{teo}\label{3.2.11 Edgar}

Sea S un espacio métrico, sean $\forall$ i $\in \mathbb{N}$ los conjuntos $F_i \subseteq S$ cerrados de S con Cov($F_i$)$\leq 1$, entonces Cov($\cup_{i \in \mathbb{N}} F_i$)$\leq 1$
\end{teo}

\begin{proof}
Debemos probar que la unión numerable de conjuntos cerrados con dimensión Cov menor o igual que uno tiene también dimensión Cov menor o igual que uno, para ello la estrategia a seguir será similar a la del teorema anterior, encontrar un cubrimiento de la unión que tenga orden menor o igual que 1. Procedamos entonces a construir ese cubrimiento. Asumiremos que S=$\cup F_i$\\ Sean los conjuntos cerrados en S, A, B, C con intersección vacía, $A\cap B \cap C= \emptyset$ entonces por (Proposición \ref{2.2.7Edgar}) existen conjuntos abiertos $A\subseteq U_0,B\subseteq V_0,C\subseteq W_0,$tales que: $$\overline{U_0}\cap \overline{V_0}\cap \overline{W_0}=\emptyset,$$ Definiremos recursivamente:

\begin{enumerate}
	\item $$\overline{U_i}\supseteq \overline{U_{i-1}},\tab \overline{V_i}\supseteq \overline{V_{i-1}},\tab \overline{W_i}\supseteq \overline{W_{i-1}},$$
	\item $$\overline{U_{i}}\cap\overline{V_{i}}\cap \overline{W_{i}}=\emptyset$$
	\item $$F_i\subseteq U_i \cup V_i \cup W_i $$

\end{enumerate}
Usaremos el procedimiento de inducción para probar que esas tres propiedades se cumplen para todo i. Si suponemos que $F_0=\emptyset$, la base de la inducción queda probada, así que ahora asumamos que 1),2) y 3) se cumplen $\forall i < k$, los conjuntos $\overline{U_{k-1}}\cap F_k$, $\overline{V_{k-1}}\cap F_k$, $\overline{W_{k-1}}\cap F_k$, por como están construidos son cerrados y con intersección vacía.Usando la Proposición \ref{PropCerrados} y  ya que Cov($F_k$)$\leq$1, existen subconjuntos de $F_k$ cerrados $K_k\supseteq \overline{U_{k-1}}\cap F_k, L_k \supseteq \overline{V_{k-1}}\cap F_k, M_k \supseteq \overline{W_{k-1}}\cap F_k$, que tienen intersección vacía y unión exactamente todo $F_k$. $K_k \cap L_k \cap M_k= \emptyset$, $F_k$, $K_k \cup L_k \cup M_k= F_k$. Ahora los conjuntos $ \overline{U_{k-1}}\cup K_k$,  $ \overline{V_{k-1}}\cup L_k$,  $ \overline{W_{k-1}}\cup M_k$, son cerrados y con intersección vacía, así que por (Proposición \ref{propAbiertos}) existen conjuntos abiertos en S:\\ $$U_k\supseteq \overline{U_{k-1}}\cup K_k , \tab V_k\supseteq \overline{V_{k-1}}\cup L_k , \tab W_k\supseteq \overline{W_{k-1}}\cup M_k, $$ tales que cubren a $F_k$ y la intersección de sus clausuras es vacía, con lo que quedaría probada la inducción.\\ Ahora si definimos los conjuntos: $$U=\bigcup_{i \in \mathbb{N}}U_i \tab V=\bigcup_{i \in \mathbb{N}}V_i \tab W=\bigcup_{i \in \mathbb{N}}W_i, $$ Por como están construidos estos conjuntos son abiertos en $\bigcup_{k \in \mathbb{N}}F_k$ lo cubren y tienen intersección vacía, por lo tanto hemos demostrado que $\bigcup_{k \in \mathbb{N}}F_k$ tiene un cubrimiento con orden 1 entonces Cov($\bigcup_{k \in \mathbb{N}}F_k)\leq 1$

\end{proof}


\begin{coro}\label{3.2.12 Edgar}

Sea S un espacio métrico, sean $\forall$ i $\in \mathbb{N}$ los conjuntos $F_i \subseteq S$ cerrados de S con Cov($F_i$)$\leq n$, entonces Cov($\cup_{i \in \mathbb{N}} F_i$)$\leq n$

\end{coro}

\begin{proof}
La prueba de este corolario es similar a la del teorema anterior con la diferencia de en vez de coger 3 conjuntos A,B,C hay que tener en cuenta n+2 conjuntos $A_1,...,A_{n+2}$ y a partir de ahí seguir los mismos pasos.
\end{proof}


\subsection{Teorema de subconjunto}

\begin{teo}\label{3.2.13 Edgar}

Sea S un espacio métrico y T un subconjunto de S, $T \subseteq S$. Entonces $Cov(T)\leq Cov(S)$
\end{teo}


\begin{proof}
Si Cov(S)=$\infty$, no hay nada que probar ya que $\infty \geq \infty$.\\Así que supongamos que Cov(S)=$n \in \mathbb{Z}$. Por lo tanto debemos ver que Cov(T)$\leq n$, para ver esto distinguiremos 3 casos, T cerrado, T abierto, T subconjunto cualquiera de S, ni cerrado ni abierto.
\begin{enumerate}
	\item Supongamos primero que T es cerrado. Sea $\mathcal{A}$ un cubrimiento abierto de T, ya que T$\subseteq$S, $\forall A \in \mathcal{A}$, $\exists E \subseteq S: A=E \cap T$, con E un abierto en S.\\ Ya que T es cerrado E$\smallsetminus$T, es abierto. Por lo tanto podemos definir un cubrimiento abierto en S de la forma: $$\mathcal{A}_1=\lbrace E \subseteq S: E \cap T\in \mathcal{A}, \,\, E \,\, abierto \rbrace \cup \lbrace S/T \rbrace ,$$
ya que Cov(S)=n, existe un refinamiento de $\mathcal{A}_1$ al que llamaremos $\mathcal{A}_2$ que tiene orden$\leq$n.\\
Ahora sea $\mathcal{A}_3=\lbrace E \cap T :E \in \mathcal{A}_2 \rbrace$, $\mathcal{A}_3$ por construcción está subordinado a $\mathcal{A}$, y es, un cubrimiento abierto de T con orden $\leq$ n, por lo tanto, Cov(T)$\leq n$.
	\item Ahora si T es abierto, T puede escribirse como la unión contable de conjuntos cerrados, $T=\bigcup_{i=1}^{\infty}F_j,$ de la forma: $$F_j=\lbrace x\in S:d(x, S/T)\geq \frac{1}{j} \rbrace ,$$ ahora por (1), $\forall j$ Cov($F_j$)$\leq$n y entonces basándonos en el corolario \ref{3.2.12 Edgar}, Cov(T)$\leq$n. 
	\item Supongamos ahora T un subconjunto general de S, no tiene por qué ser ni cerrado ni abierto, sea $\lbrace U_1,..,U_{n+2} \rbrace$ un cubrimiento abierto de T. Sean los conjuntos $V_i$, abiertos de S, tales que $\forall i$, $U_i=V_i\cap T$, ya que unión numerable de abiertos es un abierto, $$V=\bigcup_{i=1}^{n+2}V_i,$$ es un conjunto abierto en S, por (2) Cov(V)$\leq$n, entonces existe un refinamiento de $\lbrace V_1,..,V_{n+2} \rbrace$, $\lbrace W_1,..,W_{n+2} \rbrace$ tal que $\forall i$, $W_i\subseteq V_i$, $W_i$ abiertos y $$\bigcap_{i=1}^{n+2}W_i = \emptyset,\quad \bigcup_{i=1}^{n+2}W_i =V ,$$ y ya que $ W_i\cap T$ son abiertos en T, cubren a T y , $$\bigcap_{i=1}^{n+2}(W_i \cap T)=\emptyset .$$Entonces $W_i \cap T$ forman un refinamiento de orden $\leqslant$ n, y en consecuencia, podemos asegurar que Cov(T)$\leq$ n.
\end{enumerate}

\end{proof}

\section{Conjunto de cantor}
\subsection{Introducción}

El Conjunto de Cantor, también llamado en inglés "the Cantor Dust ", polvo de Cantor, dust (polvo) es el término que en inglés se utiliza para denominar a los conjuntos con dimensión topológica igual a cero.\\ Es un subconjunto de la recta real, comprendido entre el intervalo [0,1].\\ Al estar contenido en $\mathbb{R}$ es un ejemplo de fractal muy sencillo de entender y comprenderlo viene muy bien para interiorizar conceptos clave en este tipo de conjuntos, como el de autosemejanza.\\
Recursivamente se puede construir de la siguiente manera, empiezas por el intervalo [0,1]=$C_0$. Luego este lo divides en 3 partes iguales, y te quedas con los extremos derecho e izquierdo, eliminas la parte del medio. Así queda:

$$C_1=[0,\frac{1}{3}]\cup [\frac{2}{3},1].$$

Aplicaremos esta idea de dividir en 3 y quitar la parte central a los subintervalos que van quedando, así:

$$C_2=[0,\frac{1}{9}]\cup [\frac{2}{9},\frac{1}{3}]\cup [\frac{2}{3},\frac{7}{9}]\cup [\frac{8}{9},1].$$
\\
Estos conjuntos, los $C_k$ son una secuencia decreciente, es decir, $$\forall\,\,i\,\in \mathbb{N}:\quad C_i\subseteq C_{i+1},$$
\\



El  conjunto de Cantor se define como:
\begin{equation}\label{definicionCantor}
\mathcal{C}=\bigcap_{k\in \mathbb{N}}C_k 
\end{equation}


Cada conjunto $C_k$ está compuesto por $2^k$ intervalos cerrados y disjuntos entre ellos, que miden $\left( \frac{1}{3}\right)^k$ cada uno.
\\La longitud total de cada $C_k$, es la suma de las longitudes de cada intervalo disjunto que contiene, esto es: $\left( \frac{2}{3}\right)^k$.\\

Así que la longitud total de todo el conjunto de cantor será:

$$\lim_{k\longrightarrow\infty}\left( \frac{2}{3}\right)^k=0. $$\\

Que la medida 1-dimensional del cnjunto de Cantor sea 0 da a entender que la dimensión de este conjunto se va a encontrar entre 0 y 1, procedamos pues a calcular su dimensión de fraccionaria.

\begin{prop}
No hay dentro del conjunto de Cantor intervalos con longitud mayor que 0
\end{prop}
\begin{proof}
La demostración de este hecho se basa en como está construido el conjunto de Cantor, teniendo en cuenta \ref{definicionCantor}. Cada $C_k$ está formado por $2^k$ intervalos, cada uno con longitud $\frac{1}{3}^k$. Por lo tanto $\mathcal{C}$ estará formado por $\lim_{k\longrightarrow \infty}2^k=\infty$ intervalos cada uno con longitud, $\lim_{k\longrightarrow \infty}\frac{1}{3}^k=0$
\end{proof}

\subsection{Caracterización de los elementos del conjunto de Cantor}

En este apartado nos dispondremos a caracterizar los elementos del conjunto Cantor en base a su representación en base 3, lo que nos ayudará a deducir propiedades importantes del mismo, como su cardinalidad.

\begin{teo}
Sea $ x\in[0,\,1]$. Entonces x pertenece al conjunto de Cantor $ x\in\mathcal{C}$ si y solo si se puede expresar en base 3 usando solo los dígitos 0 y 2.
\end{teo}

\begin{proof}
Los números que en base 3 tienen un 1 en la primera posición decimal se encuentran dentro del intervalo:

$$[0.1_{(3},\,\, 0.122222.._{(3}]=\left[ \frac{1}{3}_{(10},\,\,\frac{2}{3}_{(10}\right]  $$

Así quedarían fuera los elementos que no están en $C_1$.\\
Notemos que los propios $ \frac{1}{3}$ y $ \frac{2}{3}$, también tienen una representación en base 3 sin utilizar el dígito "1", estas son:
$$ \frac{1}{3}_{(10}=0.02222..._{(3}, \quad \frac{2}{3}_{(10}=0.2_{(3} $$

Ahora seguimos el proceso con los números que tienen un 1 en la segunda posición decimal, que se encuentran entre 3 intervalos:
\begin{itemize}
	\item $$[0.01_{(3},\,\, 0.0122222.._{(3}]=\left[ \frac{1}{9}_{(10},\,\,\frac{2}{9}_{(10}\right],$$ Con lo que hemos eliminado los puntos del intervalo $ \left[ \frac{1}{9}_{(10},\,\,\frac{2}{9}_{(10}\right],$ menos los extremos que al igual que antes también tienen representación en base 3 sin usar el dígito 1
	\item $$[0.11_{(3},\,\, 0.1122222.._{(3}]=\left[ \frac{4}{9}_{(10},\,\,\frac{5}{9}_{(10}\right],$$ Este intervalo en el paso anterior lo hemos dejado fuera.
	\item $$[0.21_{(3},\,\, 0.2122222.._{(3}]=\left[ \frac{7}{9}_{(10},\,\,\frac{8}{9}_{(10}\right],$$ Con lo que hemos eliminado los puntos del intervalo $ \left[ \frac{7}{9}_{(10},\,\,\frac{8}{9}_{(10}\right],$ menos los extremos que al igual que antes también tienen representación en base 3 sin usar el dígito 1
\end{itemize}
Con lo que hemos eliminado los puntos que quedan fuera de $C_2$.\\
Siguiendo con esta construcción quedaría que los elementos del conjunto de Cantor son aquellos que tienen represenación en base 3 sin usar el dígito 1.
\end{proof}\\\\\\
Esta demostración es muy intuitiva viéndola desde el punto de vista de como está estructurado el conjunto de cantor. En sí este se construye empezando por el intervalo [0, 1], dividiéndolo recursivamente en 3 partes iguales y eliminando del conjunto la parte central. \\Un método gráfico para representar los números del intervalo [0, 1] en base 3, es divides el intervalo en 3 partes iguales, 

$$\left[ 0,\,\, , \frac{1}{3}\right] \cup \left[ \frac{1}{3}\,\, ,\frac{2}{3}\right]\cup \left[ \frac{2}{3}\,\, ,1\right]  $$

Si el número esta en la primera parte empezará a escribirse 0.0, si está en la segunda 0.1 y en la tercera 0.2. Luego habrá que repetir el proceso recursivamente para encontrar el resto de cifras decimales. \\
Por ejemplo si queremos representar el número 0.5 en base 3, según el método, su representación será 0.111111111... ya que por muchas veces que dividas en 3, esté siempre se va a encontrar en el centro.\\

\section{Conjuntos de Julia}

Este tipo de conjuntos, que en las condiciones óptimas se tratan de fractales, deben su nombre al matemático francés Gaston Julia, (3 de febrero de 1893, Sidi Bel Abes, Argelia - 19 de marzo de 1978, París, Francia), pionero en el estudio de fractales.\\

Este tipo e conjuntos corresponden a la frontera de atracción de los puntos fijos de un polinomio f, con grado n$\geq 2$ y coeficientes en $\mathbb{C}$ (plano complejo). En función del valor de la derivada de la función en el punto fijo, hay 3 tipos de puntos fijos, sea $z_0$ tal que f($z_0$)=$z_0$ y $\lambda$=f'($z_0$).

\begin{itemize}
	\item Si $|\lambda| >1$, el punto fijo $z_0$ es repulsor.
	\item Si $|\lambda| <1$, el punto fijo $z_0$ es atractor.
	\item Si $|\lambda| =0$, el punto fijo $z_0$ es superatractor.
	\item Si $|\lambda| =1$, el punto fijo $z_0$ es indiferente.
\end{itemize}


\begin{defi}
Sea f un polinomio con coeficientes en $\mathbb{C}$ y grado n$\geq 2$. Se define el Conjunto de Julia completo de f, K(f) como:

$$K(f)=\lbrace z \in \mathbb{C}:\quad f^k(z)\nrightarrow \infty \rbrace $$
\end{defi}

\begin{defi}
Se define el Conjunto de Julia f, J(f) como la frontera del conjunto de Julia completo, $\delta(K(f))=J(f)$.
\end{defi}

Por lo tanto si $z\in J(f)$, en todo entorno de z hay puntos v y w tales que, 

$$f^k(w)\longrightarrow \infty, \quad f^k(v)\nrightarrow \infty. $$

\begin{ej}
Sea la función, f:$\mathbb{C}\longrightarrow \mathbb{C}$, tal que $f(z)=z^2$. se tiene que:

$$f^k(z)=z^{2^k}, $$

por lo tanto:

\begin{itemize}
	\item Si $|z|<1$, $\Rightarrow \quad f^k(z)\longrightarrow 0.$
	\item Si $|z|=1$, cuando k tienda a infinito, z prebalecerá en la circunferencia de radio 1.
	\item Si $|z|>1$, $\Rightarrow \quad f^k(z)\longrightarrow \infty.$

\end{itemize}
Se puede ver que $f(z)=z^2$, tiene 2 puntos fijos, z=0 y z=1, pero mientras que el 0 es superatractor, $f'(0)=0$, el 1 es repulsor $f'(1)=2$.

\subsection{Conjunto de Julia para la familia cuadrática}

Llamaremos familia al conjunto de funciones con la forma:

$$f_c(z)=z^2+c, \quad c\in\mathbb{C}. $$

Estas familias de funciones tienen dos puntos fijos,

\begin{itemize}
	\item $z_0=\frac{1+\sqrt{1-4c}}{2},$
	\item $z_1=\frac{1-\sqrt{1-4c}}{2}.$
\end{itemize}

Ya que $f_c'(z)=2z$, se tiene que $|f_c'(z_0)|>1$ y que $|f_c'(z_1)|<1$, por lo tanto $z_0$ es repulsor y $z_1$ atractor.

\begin{prop}
El conjunto de Julia completo para la familia cuadrática de funciones está incluido en la bola de centro en 0 y radio máximo entre $|c|$ y 2.
\end{prop}
\begin{proof}
Tenemos que probar que si $|z|>max\lbrace 2,|c| \rbrace $ entonces: $$f_c^k(z)_{k\longrightarrow \infty}\longrightarrow \infty.$$
Sea $\epsilon >0$ y z$\in \mathbb{C}$ tal que, $|z|>2+\epsilon$.\\
$$|f_c(z)|=|z^2+c|\geqslant |z^2|-|c|\geqslant |z^2|-|z|=|z|(|z|-1) $$
ya que $|z|>2+\epsilon \Rightarrow |z|-1>1+\epsilon$, por lo tanto,
$$|z|(|z|-1)\geqslant |z|(1+\epsilon) \Rightarrow |f_c(z)| \geqslant |z|(1+\epsilon),\quad |f_c^k(z)| \geqslant |z|(1+\epsilon)^k\longrightarrow \infty $$
\end{proof}
